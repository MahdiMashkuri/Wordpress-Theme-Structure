wordpress-theme-structure/
│
├── style.css                              # فایل اصلی استایل، شامل اطلاعات متا و استایل‌های اصلی قالب
├── index.php                              # فایل اصلی قالب که برای نمایش محتوای اصلی استفاده می‌شود
├── functions.php                          # فایل برای تعریف توابع و پیکربندی‌های ویژه قالب
├── header.php                             # فایل برای نمایش بخش هدر سایت
├── footer.php                             # فایل برای نمایش بخش فوتر سایت
├── sidebar.php                            # فایل برای نمایش نوار کناری (سایدبار)
├── screenshot.png                         # تصویر بندانگشتی قالب برای نمایش در داشبورد وردپرس
├── rtl.css                                # استایل‌های راست‌چین برای زبان‌های RTL
├── PROJECT-STRUCTURE-fa.txt               # فایل توضیحات ساختار قالب وردپرس
├── README.md                              # فایل توضیحات قالب برای مستندات یا مخزن وردپرس
│
├── assets/                                # پوشه شامل منابع استاتیک
│   ├── css/                               # پوشه برای فایل‌های استایل
│   │   ├── main.css                       # استایل‌های عمومی قالب
│   │   ├── admin.css                      # استایل‌های بخش مدیریت
│   │   ├── editor-style.css               # استایل‌های ویرایشگر گوتنبرگ
│   │   └── custom.css                     # استایل‌های سفارشی اضافه
│   ├── js/                                # پوشه برای فایل‌های جاوااسکریپت
│   │   ├── main.js                        # اسکریپت‌های عمومی قالب
│   │   ├── admin.js                       # اسکریپت‌های بخش مدیریت
│   │   ├── navigation.js                  # اسکریپت برای مدیریت منوی ناوبری
│   │   ├── customizer.js                  # اسکریپت برای سفارشی‌سازی از طریق Customizer
│   │   └── custom.js                      # اسکریپت‌های سفارشی اضافه
│   ├── images/                            # پوشه برای تصاویر قالب
│   │   ├── logo.png                       # لوگوی سایت
│   │   ├── background.jpg                 # تصویر پس‌زمینه
│   │   ├── banner.jpg                     # بنرهای تبلیغاتی یا معرفی
│   │   └── placeholder.png                # تصویر پیش‌فرض برای محتواهای بدون تصویر
│   └── fonts/                             # پوشه برای فونت‌های سفارشی
│       ├── custom-font.ttf                # فونت TrueType
│       ├── custom-font.woff               # فونت Web Open Font Format
│       └── custom-font.woff2              # فونت Web Open Font Format 2
│
├── templates/                             # پوشه شامل فایل‌های الگو
│   ├── parts/                             # پوشه برای بخش‌های قالب مانند هدر، فوتر و سایدبار
│   │   ├── header.php                     # بخش هدر
│   │   ├── footer.php                     # بخش فوتر
│   │   ├── sidebar.php                    # بخش سایدبار
│   │   ├── content.php                    # بخش محتوا
│   │   ├── pagination.php                 # بخش ناوبری صفحات (پاجینیشن)
│   │   ├── breadcrumbs.php                # بخش Breadcrumbs
│   │   └── comments.php                   # بخش نظرات
│   ├── pages/                             # پوشه برای قالب‌های سفارشی صفحات
│   │   ├── template-home.php              # قالب سفارشی برای صفحه اصلی
│   │   ├── template-about.php             # قالب سفارشی برای صفحه درباره ما
│   │   └── template-contact.php           # قالب سفارشی برای صفحه تماس با ما
│   ├── single/                            # پوشه برای قالب‌های پست‌های تکی
│   │   ├── single-post.php                # قالب نمایش پست تکی
│   │   ├── single-product.php             # قالب نمایش محصول تکی (برای سایت‌های فروشگاهی)
│   │   └── single-custom.php              # قالب نمایش پست سفارشی تکی
│   ├── archive/                           # پوشه برای قالب‌های آرشیو
│   │   ├── archive.php                    # قالب نمایش آرشیو عمومی
│   │   ├── category.php                   # قالب نمایش دسته‌بندی‌ها
│   │   ├── tag.php                        # قالب نمایش برچسب‌ها
│   │   ├── author.php                     # قالب نمایش نوشته‌های نویسنده
│   │   └── date.php                       # قالب نمایش آرشیو بر اساس تاریخ
│   ├── search/                            # پوشه برای قالب‌های جستجو
│   │   └── search.php                     # قالب نمایش نتایج جستجو
│   └── 404.php                            # قالب برای نمایش صفحه 404 (صفحه‌ای که پیدا نشد)
│
├── includes/                              # پوشه شامل فایل‌های PHP برای منطق و توابع قالب
│   ├── customizer/                        # فایل‌های مرتبط با سفارشی‌سازی قالب
│   │   ├── customizer.php                 # تنظیمات و توابع مرتبط با Customizer
│   │   ├── controls/                      # کنترل‌های سفارشی برای Customizer
│   │   ├── sanitize.php                   # توابع پاکسازی و اعتبارسنجی برای Customizer
│   │   └── sections.php                   # بخش‌های سفارشی در Customizer
│   ├── hooks/                             # هوک‌ها و اکشن‌های سفارشی قالب
│   │   ├── actions.php                    # اکشن‌ها
│   │   ├── filters.php                    # فیلترها
│   │   └── ajax.php                       # مدیریت درخواست‌های AJAX
│   ├── classes/                           # پوشه برای کلاس‌های سفارشی قالب
│   │   ├── class-theme-setup.php          # کلاس برای تنظیمات اولیه قالب
│   │   ├── class-walker-nav.php           # کلاس برای ناوبری سفارشی
│   │   ├── class-custom-post-type.php     # کلاس برای مدیریت انواع پست سفارشی
│   │   ├── class-meta-boxes.php           # کلاس برای مدیریت متاباکس‌های سفارشی
│   │   ├── class-shortcodes.php           # کلاس برای مدیریت شورت‌کدها
│   │   └── class-widgets.php              # کلاس برای مدیریت ویجت‌های سفارشی
│   ├── helpers/                           # پوشه برای توابع و ابزارهای کمکی
│   │   ├── template-tags.php              # توابع مربوط به برچسب‌های قالب
│   │   ├── utils.php                      # توابع کمکی عمومی
│   │   ├── extras.php                     # توابع اضافه برای قابلیت‌های سفارشی
│   │   └── security.php                   # توابع مرتبط با امنیت قالب
│   ├── setup/                             # پوشه برای تنظیمات و راه‌اندازی قالب
│   │   ├── theme-support.php              # افزودن پشتیبانی از قابلیت‌های مختلف وردپرس (مثل تصویر شاخص)
│   │   ├── enqueue-scripts.php            # صف بندی اسکریپت‌ها و استایل‌ها
│   │   ├── menu-register.php              # ثبت و مدیریت منوها
│   │   ├── widget-areas.php               # مدیریت ناحیه‌های ویجت
│   │   ├── editor.php                     # پیکربندی ویرایشگر بلوک (گوتنبرگ)
│   │   └── theme-setup.php                # تنظیمات اولیه قالب (مثل ترجمه‌ها)
│   ├── shortcodes/                        # پوشه برای شورت‌کدهای قالب
│   │   ├── shortcode-example.php          # نمونه‌ای از یک شورت‌کد
│   │   └── shortcode-another.php          # یک شورت‌کد دیگر
│   └── custom-post-types/                 # پوشه برای تعریف انواع پست سفارشی
│       ├── cpt-portfolio.php              # نوع پست سفارشی "پورتفولیو"
│       └── cpt-testimonials.php           # نوع پست سفارشی "نظرات مشتریان"
│
├── languages/                             # پوشه شامل فایل‌های ترجمه (PO و MO)
│   ├── en_US.po                           # فایل ترجمه انگلیسی
│   ├── en_US.mo                           # فایل باینری ترجمه انگلیسی
│   ├── fa_IR.po                           # فایل ترجمه فارسی
│   └── fa_IR.mo                           # فایل باینری ترجمه فارسی
│
├── vendor/                                # پوشه برای وابستگی‌های Composer (در صورت استفاده از Composer)
│   ├── autoload.php                       # فایل بارگذاری خودکار Composer
│   ├── composer.json                      # فایل پیکربندی Composer
│   └── composer.lock                      # فایل قفل Composer برای نسخه‌های وابستگی
│
├── tests/                                 # پوشه شامل فایل‌های تست
│   ├── test-theme.php                     # تست‌های اصلی قالب
│   ├── bootstrap.php                      # راه‌اندازی اولیه برای تست‌ها
│   ├── phpunit.xml                        # فایل پیکربندی PHPUnit
│   └── test-helpers.php                   # توابع و ابزارهای کمکی برای تست‌ها
│
├── .gitignore                             # فایل برای تنظیمات Git به منظور نادیده گرفتن فایل‌های خاص
└── CHANGELOG.md                           # فایل برای ثبت تغییرات و نسخه‌های جدید قالب
